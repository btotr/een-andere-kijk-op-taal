\startuseMPgraphic{cover}
   StartPage ;
     fill Page withcolor yellow ;
      path p ; p := unitcircle randomized .03 xsized(PaperWidth-1cm) ;
     fill p shifted -center p shifted center Page  withcolor red;
   StopPage ;
\stopuseMPgraphic

\defineoverlay[cover][\useMPgraphic{cover}]
\setupbackgrounds[page][background=cover]

  \raggedcenter
  \vfill \definedfont[file:Roboto-Regular.otf at 36pt]\setstrut \strut Een andere kijk op taal 
  \blank[2*big] \definedfont[file:Roboto-Regular.otf at 10pt]\setstrut \strut Colin Meerveld
  \vfill
\stopstandardmakeup
